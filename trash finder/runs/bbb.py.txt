import cv2

cap = cv2.VideoCapture(0, cv2.CAP_DSHOW)

cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1280)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 720)

print("Camera Open:", cap.isOpened())

while True:
    ret, frame = cap.read()

    if not ret:
        print("프레임 가져오기 실패")
        break

    cv2.imshow("Camera Test", frame)

    key = cv2.waitKey(1)

    if key == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
